const map = L.map('map').setView([20, 0], 2); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom: 19 }).addTo(map); let shapes = []; let mergedLayer = null; function toKm(miles) { return miles * 1.60934; } function rebuild() { if (mergedLayer) { map.removeLayer(mergedLayer); mergedLayer = null; } if (shapes.length === 0) return; // 🔥 SAFE STRATEGY: rebuild from scratch each time let result = shapes[0]; for (let i = 1; i < shapes.length; i++) { try { const merged = turf.union(result, shapes[i]); if (merged) result = merged; } catch (e) { console.log("Skipping invalid union"); } } mergedLayer = L.geoJSON(result, { style: { color: 'red', fillColor: 'red', fillOpacity: 0.3 } }).addTo(map); } map.on('click', (e) => { const miles = parseFloat(document.getElementById('radiusInput').value); const mode = document.getElementById('modeSelect').value; if (!miles || miles <= 0) return; const circle = turf.circle( [e.latlng.lng, e.latlng.lat], toKm(miles), { steps: 64, units: 'kilometers' } ); if (mode === "outside") { shapes.push(circle); rebuild(); } else { L.circle(e.latlng, { radius: toKm(miles) * 1000, color: 'blue', fillColor: 'blue', fillOpacity: 0.3 }).addTo(map); } }); // clear button document.getElementById("clearBtn").addEventListener("click", () => { shapes = []; if (mergedLayer) map.removeLayer(mergedLayer); });
